class Dijkstra_Node:
    def __init__(self, name, cost, adjacent_hubs):
        self.visited = False
        self.name = name
        self.true_cost = cost
        self.relative_cost = float("inf")
        self.origin = None
        self.links = adjacent_hubs
        self.time = 0

def do_path(work_hub, start):
    output = []
    while (work_hub.name != start):
        output.append((work_hub.name, work_hub.time))
        work_hub = work_hub.origin
        # print(work_hub)
    # print(output)
    return output




def do_reservation(work_hub, start, reserved, available, end):
    while work_hub.name != start:
        work_hub = work_hub.origin
        reserved.append(tuple((work_hub.name, work_hub.time)))
    return reserved
